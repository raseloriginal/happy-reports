Database Schema
Production-ready · Normalized · MongoDB collections with SQL-style logic

Core Collections / Tables

-- COMPANIES
CREATE TABLE companies (
  id UUID PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  contact_person VARCHAR(100),
  phone VARCHAR(20),
  address TEXT,
  credit_limit DECIMAL(15,2),
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP
);

-- WAREHOUSES
CREATE TABLE warehouses (
  id UUID PRIMARY KEY,
  name VARCHAR(100),
  location VARCHAR(200),
  capacity_sqft INT,
  manager_id UUID REFERENCES users(id),
  is_active BOOLEAN,
  created_at TIMESTAMP
);

-- PRODUCTS
CREATE TABLE products (
  id UUID PRIMARY KEY,
  name VARCHAR(200),
  sku VARCHAR(50) UNIQUE,
  company_id UUID REFERENCES companies(id),
  category VARCHAR(50),
  unit VARCHAR(20),
  purchase_price DECIMAL(12,2),
  sale_price DECIMAL(12,2),
  min_stock_level INT,
  is_active BOOLEAN
);

-- INVENTORY LOTS
CREATE TABLE inventory_lots (
  id UUID PRIMARY KEY,
  lot_number VARCHAR(50) UNIQUE,
  product_id UUID REFERENCES products(id),
  warehouse_id UUID REFERENCES warehouses(id),
  company_id UUID REFERENCES companies(id),
  quantity INT NOT NULL,
  received_date DATE,
  expiry_date DATE,
  batch_code VARCHAR(50),
  status ENUM('active','depleted','expired'),
  cost_price DECIMAL(12,2)
);

-- INVENTORY TRANSACTIONS
CREATE TABLE inventory_transactions (
  id UUID PRIMARY KEY,
  lot_id UUID REFERENCES inventory_lots(id),
  warehouse_id UUID REFERENCES warehouses(id),
  txn_type ENUM('stock_in','sale','damage','expiry','transfer','return','adjustment'),
  quantity INT,
  reference_id UUID,
  note TEXT,
  created_by UUID REFERENCES users(id),
  created_at TIMESTAMP
);

-- COMPANY LEDGER
CREATE TABLE company_ledger (
  id UUID PRIMARY KEY,
  company_id UUID REFERENCES companies(id),
  txn_type ENUM('deposit','lot_received','credit_note','debit_note','return','adjustment'),
  amount DECIMAL(15,2) NOT NULL,
  reference_no VARCHAR(100),
  payment_method ENUM('bank','bkash','nagad','cash','cheque'),
  bank_name VARCHAR(100),
  description TEXT,
  running_balance DECIMAL(15,2),
  created_by UUID REFERENCES users(id),
  created_at TIMESTAMP
);

-- SALES
CREATE TABLE sales (
  id UUID PRIMARY KEY,
  invoice_no VARCHAR(50) UNIQUE,
  retailer_id UUID REFERENCES retailers(id),
  warehouse_id UUID REFERENCES warehouses(id),
  sr_id UUID REFERENCES users(id),
  total_amount DECIMAL(15,2),
  discount DECIMAL(12,2),
  net_amount DECIMAL(15,2),
  collection_status ENUM('paid','pending','partial','overdue'),
  sale_date DATE,
  created_at TIMESTAMP
);

-- EXPENSES
CREATE TABLE expenses (
  id UUID PRIMARY KEY,
  category ENUM('salary','logistics','fuel','rent','electricity','internet','mobile','office','misc','vehicle_maintenance'),
  amount DECIMAL(12,2),
  warehouse_id UUID REFERENCES warehouses(id),
  description TEXT,
  voucher_no VARCHAR(50),
  expense_date DATE,
  approved_by UUID REFERENCES users(id),
  created_at TIMESTAMP
);

-- INVESTORS
CREATE TABLE investors (
  id UUID PRIMARY KEY,
  name VARCHAR(100),
  phone VARCHAR(20),
  capital_invested DECIMAL(15,2),
  profit_share_percent DECIMAL(5,2),
  investment_date DATE,
  is_active BOOLEAN
);

-- INVESTOR TRANSACTIONS
CREATE TABLE investor_transactions (
  id UUID PRIMARY KEY,
  investor_id UUID REFERENCES investors(id),
  txn_type ENUM('investment','profit_distribution','withdrawal'),
  amount DECIMAL(15,2),
  description TEXT,
  txn_date DATE
);

-- RETAILERS
CREATE TABLE retailers (
  id UUID PRIMARY KEY,
  name VARCHAR(200),
  owner_name VARCHAR(100),
  phone VARCHAR(20),
  upazila VARCHAR(100),
  address TEXT,
  sr_id UUID REFERENCES users(id),
  credit_limit DECIMAL(12,2),
  is_active BOOLEAN
);

-- USERS & PERMISSIONS
CREATE TABLE users (
  id UUID PRIMARY KEY,
  name VARCHAR(100),
  phone VARCHAR(20),
  email VARCHAR(100),
  role ENUM('super_admin','ceo','finance_manager','warehouse_manager','sales_manager','sr','viewer'),
  warehouse_id UUID REFERENCES warehouses(id),
  is_active BOOLEAN,
  last_login TIMESTAMP
);

-- AUTO-CALCULATED VIEWS
CREATE VIEW company_balance AS
  SELECT
    c.id,
    c.name,
    COALESCE(SUM(CASE WHEN cl.txn_type='deposit' THEN cl.amount ELSE 0 END), 0) AS total_deposit,
    COALESCE(SUM(CASE WHEN cl.txn_type='lot_received' THEN cl.amount ELSE 0 END), 0) AS total_received,
    COALESCE(SUM(CASE WHEN cl.txn_type='deposit' THEN cl.amount ELSE 0 END), 0) -
    COALESCE(SUM(CASE WHEN cl.txn_type='lot_received' THEN cl.amount ELSE 0 END), 0) AS balance
  FROM companies c
  LEFT JOIN company_ledger cl ON c.id = cl.company_id
  GROUP BY c.id, c.name;
